<template>
    <div>
        <h1>Sesiones</h1>
        <table>
            <thead>
                <tr>
                    <th>ID</th>
                    <th>Nombre</th>
                    <th>Fecha</th>
                    <th>Duración</th>
                    <th>Estado</th>
                </tr>
            </thead>
            <tbody>
                <tr v-for="sesion in sesiones" :key="sesion._id">
                    <td>{{ sesion._id }}</td>
                    <td>{{ sesion.nombre }}</td>
                    <td>{{ sesion.fecha }}</td>
                </tr>
            </tbody>
        </table>
    </div>
</template>

<script>
import axios from 'axios'

export default {
    data() {
        return {
            sesiones: []
        }
    },
    created() {
        this.fetchSesiones()
    },
    methods: {
        async fetchSesiones() {
            try {
                const response = await axios.get('http://localhost:8080/sesion')
                this.sesiones = response.data
            } catch (error) {
                console.error('Error fetching sesiones:', error)
            }
        }
    }
}
</script>
